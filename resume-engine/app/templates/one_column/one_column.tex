\newcommand{\resumeEntry}[3]{
    \begin{tabular*}{\textwidth}{l@{\extracolsep{\fill}}r}
        \textbf{#1} & \textit{\small #3} \\
    \end{tabular*}
    \\ \hspace{0.1in} \textit{#2}
}

\newcommand{\resumeDescription}[1]{
    \begin{itemize}[leftmargin=0.2in, itemsep=0pt, label={$\circ$}]
        \item\small #1
    \end{itemize}
}
